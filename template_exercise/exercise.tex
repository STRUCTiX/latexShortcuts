\begin{luacode*}
function exercisetable(n)
        tex.print('\\begin{table}\\begin{tabular}{')
        for i = 0, n-1 do
                tex.print('c|')
        end
        tex.print('c}')
        for i = 0, n-1 do
                tex.print(string.format('%d \\hspace{1cm} &', i + 1))
        end
        tex.print('$\\sum$ \\hspace{1cm} \\\\ \\hline')
        for i = 0, n-1 do
                tex.print('&')
        end
        tex.print('\\\\ \\end{tabular}\\end{table}')
end
\end{luacode*}
